#include <stdio.h>

int find_max_element(int *ptr, int n) {
    int max = *ptr;
    for (int i = 0; i < n; i++) {
        if (*(ptr + i) > max) {
            max = *(ptr + i);
        }
    }
    return max;
}

    int arr[100];
    for (int i = 0; i < n; i++) {

    }

}
